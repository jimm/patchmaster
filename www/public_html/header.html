  zone C0..B3
  xpose 12
end

conn :mb, 1, :kzwl, 1 do
  zone C4..B8
  filter do |conn, bytes|
    if bytes.note?
      bytes[2] *= 2
      bytes[2] &= 0x7f
    end
    bytes
  end
end
