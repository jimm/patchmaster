conn :mb, 1, :d4, 10 do
  zone C0..B3
  xpose 12
end

conn :mb, 1, :kzwl, 1 do
  zone C4..B8
  filter do |c,b|
    if b.note?
      b[2] *= 2
      b[2] &= 0x7f
    end
    b
  end
end
